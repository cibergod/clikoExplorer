rsprav.db - знавание справочников которые используються
rfdef.db -ссылки на другие формы вроде 
rekw - парметры полей формы справочника
rdef - сопоставление формы справочника
rcdef - какието параметры пока фиг его значет 
rddef - какието сопоставления пока тоже фиг его знает 

KLIKOCFG - спиок форм в конфигурации
Интересные поля 

//данные о конфигурации 
SELECT    
IDR, /*уникальный номер */    
NAME,  /*имя формы*/
RCLASS,  /*Уникальный номер формы*/
SNAME,  /*пткаталог для работы формы*/
BEGDATE, /*дада использования формы*/
VER_DATE /*Дата обновления */
FROM            
KLIKOCFG /*файл настройки Clicko*/
